html

pug

+textLists({
  type:'dot',
  lists:[
    {
      text:"블라블라~블라블라~블라블라~블라블라~",
      className:"custom_class_name",
      subLists:{
        type:'hyphhen',
        lists:[
          {
            text:"가나다라마바사아자차카타파하-sub"
          },
          {
            text:"가나다라마바사아자차카타파하-sub",
          },
          {
            text:"가나다라마바사아자차카타파하-sub"
          },
        ]
      }
    },
    {
      text:"블라블라~블라블라~블라블라~블라블라~"
    },
    {
      text:"블라블라~블라블라~블라블라~블라블라~블라블라~"
    },
  ]
})


Attribute

Attributes
Property Description Type Default
